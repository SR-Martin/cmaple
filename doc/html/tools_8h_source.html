1045 template<class T1, class T2>
1046 void quicksort(T1* arr, int left, int right, T2* arr2 = NULL) {
1047 if (left > right) return;
1048 ASSERT(left <= right);
1049 int i = left, j = right;
1050 T1 pivot = arr[(left + right) / 2];
1051
1052 /* partition */
1053 while (i <= j) {
1054 while (arr[i] < pivot)
1055 ++i;
1056 while (arr[j] > pivot)
1057 --j;
1058 if (i <= j) {
1059 T1 tmp = arr[i];
1060 arr[i] = arr[j];
1061 arr[j] = tmp;
1062 if (arr2) {
1063 T2 tmp2 = arr2[i];
1064 arr2[i] = arr2[j];
1065 arr2[j] = tmp2;
1066 }
1067 ++i;
1068 --j;
1069 }
1070 };
1071
1072 /* recursion */
1073 if (left < j)
1074 quicksort(arr, left, j, arr2);
1075 if (i < right)
1076 quicksort(arr, i, right, arr2);
1077 }
